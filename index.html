css3 lightbox

about

CSS3 lightbox, using no images, and no CSS files. For more info, see the github repository

examples

text lightbox 
$('#text_lightbox').click(function() {
  $.lightbox({type: 'text', text: 'Hey there! This is a lightbox'});
});
ajax lightbox 
$('#ajax_lightbox').click(function() {
  $.lightbox({ type: 'ajax', url: 'ajax.html' });
});
div lightbox 
$('#div_lightbox').click(function() {
  $.lightbox({ type: 'div', element: $('#lightbox_content')});
});
custom lightbox 
$('#options_lightbox').click(function() {
  $.lightbox({
    type: 'div', 
    element: $('#lightbox_content'),
    css: {
      shadow: {
        'background-color': 'white',
        'opacity': 0.5
      },
      modal: {
        'background-color': 'white',
        'color': 'black',
        'border': '5px solid black'
      }
    }
  });
});
